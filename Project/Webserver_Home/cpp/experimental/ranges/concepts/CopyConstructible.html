template <class T>

concept bool CopyConstructible() {
    return ranges::MoveConstructible<T>() &&
           ranges::Constructible<T, const std::remove_cv_t<T>&>() &&
           ranges::ConvertibleTo<std::remove_cv_t<T>&, T>() &&
           ranges::ConvertibleTo<const std::remove_cv_t<T>&, T>() &&
           ranges::ConvertibleTo<const std::remove_cv_t<T>&&, T>();

}
